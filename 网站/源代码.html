#include<stdio.h>
void aver(float (*p)[5]);
void search_1(float (*p)[5]);
void search_2(float (*p)[5]);
int main(void)
{
    float score[4][5];
    int i;
    printf("请输入四个学生各五门课的成绩，用空格分开：\n");
    for(i=0;i<20;i++) scanf("%f",score[0]+i);
    aver(score);
    printf("\n");
    search_1(score);
    printf("\n");
    search_2(score);
    return 0;
}
void aver(float (*p)[5])
{
    int i;
    float sum;
    for(i=0,sum=0;i<4;i++)  sum=sum+*(*(p+i));
    printf("The average of NO.1 subject is %5.2f\n",sum/4.0);
}
void search_1(float (*p)[5])
{
    int i,j,num;
    float sum;
    for(i=0;i<4;i++)
    {
        for(j=0,num=0;j<5;j++)
        {
            if(*(*(p+i)+j)<60) num=num+1;
        }
        if(num>=2)
        {
            printf("NO.%d student fail\nhis scores are:",i+1);
            for(j=0,sum=0;j<5;j++)
            {
                printf("%5.2f ",*(*(p+i)+j));
                sum=sum+*(*(p+i)+j);
            }
            printf("\nhis average is %5.2f\n",sum/5.0);
        }
    }
}
void search_2(float (*p)[5])
{
    int i,j,num;

    for(i=0;i<4;i++)

        for(j=0,sum=0,num=0;j<5;j++)

            sum=sum+*(*(p+i)+j);
